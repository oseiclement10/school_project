bubble sort
#include <iostream>

using namespace std;

int main(){

int n,temp;
cout<<"enter size of your array"<<endl;
cin>>n;
int numbers[n];
cout<<"enter numbers"<<endl;
for(int i=0;i<n;i++){
    cin>>numbers[i];
}
cout<<"your numbers before the bubble sort \n";
for(int i=0;i<n;i++){
    cout<<numbers[i]<<",";
}

for(int i=0;i<n-1;i++){
    for(int j=0;j<n-i-1;j++){
        if (numbers[j]>numbers[j+1]){
            temp = numbers[j];
            numbers[j]=numbers[j+1];
            numbers[j+1] = temp;
        }
    }
}

cout<<"\n";
cout<<"here is your array after the bubble sort \n";
for(int i=0;i<n;i++){
    cout<<numbers[i]<<",";
}
}

selection sort
#include<iostream>;
using namespace std;

int main(){
int n;
cout<<"enter size of array"<<endl;
cin>>n;
int elements[n];
//take in array elements

for(int i=0;i<n;i++){
    cout<<"enter value for elem "<< i+1 <<endl;
    cin>>elements[i];
}

//lets sort
int minn;
int temp;

for(int i=0;i<n;i++){
    minn = elements[i];
    temp = i;
    for(int j=i+1;j<n;j++){
        if(minn>elements[j]){
           minn = elements[j];
           temp = j;
        }
    }
    elements[temp]=elements[i];
    elements[i]=minn;

}

cout<<"results after sorting"<<endl;
for(int i=0;i<n;i++){
    cout<<elements[i]<<" ";
}
}
insertion sort
#include <iostream>
using namespace std;
int main(){
int n,j,temp,key;
cout<<"enter array size"<<endl;
cin>>n;
int arr[n];
for (int i=0;i<n;i++){
    cout<<"enter value of elem at position "<<i+1<<endl;
    cin>>arr[i];
}

//lets sort;

for(int i=1;i<n;i++){
    key = arr[i];
    int j = i-1;
    while(j>=0&&key<arr[j]){
        temp = arr[j];
        arr[j] = arr[j+1];
        arr[j+1] = temp;
        j--;
    }
}
cout<<"\n"<<endl;
cout<<"results after sorting"<<endl;
for(int i=0;i<n;i++){
    cout<<arr[i]<<",";
}

}
